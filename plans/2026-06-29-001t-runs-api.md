# Runs Backend — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship run CRUD, launch, execution engine (outer + inner loop), checkpoint writes, LookupDeltaCR interrupt, and resume against existing `RunRecord` / `RunCheckpoint` / `MappingSnapshot` / `LookupSnapshot` ORM models.

**Architecture:** Synchronous execution in the launch request (no background queue). The outer loop pins approved snapshot versions onto `RunRecord`; the inner loop applies `MappingSnapshot.field_bindings` then `LookupSnapshot.value_map` row by row, writes checkpoints every 500 rows, pauses on unmapped values via `LookupDeltaCR`. Resume restores from checkpoint row offset.

**Depends on:** 001q (SourceSlice + SourceSliceRow), 001r (CodeGenerationArtifact + migration 0009), 001s (AI adapter — not used in this plan but must be importable).

**Tech Stack:** Python ≥ 3.11, FastAPI, SQLAlchemy 2.x sync ORM, Pydantic v2, Alembic, pytest + FastAPI TestClient.

## Global Constraints

- `from __future__ import annotations` at top of every file
- All IDs: UUID4 via `new_id()` from `db.models`
- Every mutation calls `record_management_audit()` from `management.platform`
- Tests hit real SQLite test DB via FastAPI TestClient — no mocking
- Run statuses: `"queued"` | `"running"` | `"paused"` | `"completed"` | `"failed"` | `"awaiting_approval"`
- Checkpoint interval: 500 rows (constant `CHECKPOINT_INTERVAL = 500`)
- I18: every DDL change ships with a hand-written migration; read `down_revision` from the actual file

```python
from __future__ import annotations

from sqlalchemy.orm import Session

from ..db.models import (
    MappingArtifact, MappingSnapshot, LookupSnapshot,
    RunRecord, SourceSlice, SourceSliceRow, new_id,
)
from .checkpoints import restore_checkpoint, write_checkpoint
from .inner_loop import process_rows
from .lookup_delta import raise_lookup_delta_cr


class MissingApprovalError(Exception):
    pass


def pin_snapshots(run: RunRecord, db: Session) -> None:
    slice_ = (
        db.query(SourceSlice)
        .filter_by(source_definition_id=run.source_definition_reference, status="approved")
        .order_by(SourceSlice.created_at.desc())
        .first()
    )
    if slice_ is None:
        raise MissingApprovalError("no approved source slice for this source definition")

    mapping = (
        db.query(MappingSnapshot)
        .filter_by(project_id=run.project_id,
                   destination_object_name=run.destination_object_name,
                   status="approved")
        .order_by(MappingSnapshot.created_at.desc())
        .first()
    )
    if mapping is None:
        raise MissingApprovalError("no approved mapping snapshot for this destination object")

    run.source_slice_version = slice_.source_slice_version
    run.mapping_snapshot_version = mapping.mapping_snapshot_version
    db.flush()


def execute_run(run_id: str, db: Session) -> None:
    run = db.get(RunRecord, run_id)
    if run is None:
        raise ValueError(f"run {run_id} not found")

    run.status = "running"
    run.current_stage = "row_mapping"
    db.flush()

    pin_snapshots(run, db)

    mapping = (
        db.query(MappingSnapshot)
        .filter_by(project_id=run.project_id,
                   destination_object_name=run.destination_object_name,
                   mapping_snapshot_version=run.mapping_snapshot_version)
        .one()
    )

    start_row = restore_checkpoint(run, db)

    rows = (
        db.query(SourceSliceRow)
        .join(SourceSlice, SourceSliceRow.source_slice_id == SourceSlice.source_slice_id)
        .filter(SourceSlice.source_slice_version == run.source_slice_version)
        .filter(SourceSliceRow.row_index >= start_row)
        .order_by(SourceSliceRow.row_index)
        .all()
    )

    artifact = MappingArtifact(
        mapping_artifact_id=new_id(),
        run_id=run.run_id,
        project_id=run.project_id,
        destination_object_name=run.destination_object_name,
        mapping_snapshot_version=run.mapping_snapshot_version,
        lookup_snapshot_version=run.lookup_snapshot_version or "",
        mapped_rows=[],
    )
    db.add(artifact)
    db.flush()

    try:
        process_rows(run, rows, mapping, artifact, db)
    except _LookupDeltaNeeded as exc:
        raise_lookup_delta_cr(run, exc.lookup_name, exc.source_value, db)
        return

    run.status = "completed"
    run.completion_metadata = {"mapped_row_count": len(artifact.mapped_rows)}
    db.commit()


class _LookupDeltaNeeded(Exception):
    def __init__(self, lookup_name: str, source_value: str) -> None:
        self.lookup_name = lookup_name
        self.source_value = source_value
```

```python
from __future__ import annotations

from sqlalchemy.orm import Session

from ..db.models import RunCheckpoint, RunRecord, new_id

CHECKPOINT_INTERVAL = 500


def write_checkpoint(run: RunRecord, row_index: int, db: Session) -> None:
    cp = RunCheckpoint(
        run_checkpoint_id=new_id(),
        run_id=run.run_id,
        current_stage=run.current_stage,
        current_object=run.destination_object_name,
        current_environment=run.environment,
        approved_snapshots={
            "source_slice_version": run.source_slice_version,
            "mapping_snapshot_version": run.mapping_snapshot_version,
            "lookup_snapshot_version": run.lookup_snapshot_version,
        },
        last_completed_checkpoint_boundary=str(row_index),
        checkpoint_payload={"row_index": row_index},
    )
    db.add(cp)
    db.flush()


def restore_checkpoint(run: RunRecord, db: Session) -> int:
    latest = (
        db.query(RunCheckpoint)
        .filter_by(run_id=run.run_id)
        .order_by(RunCheckpoint.created_at.desc())
        .first()
    )
    if latest is None or latest.last_completed_checkpoint_boundary is None:
        return 0
    return int(latest.last_completed_checkpoint_boundary)
```

```python
from __future__ import annotations

import csv
import io
from typing import Any

from sqlalchemy.orm import Session

from ..db.models import MappingArtifact, MappingSnapshot, LookupSnapshot, RunRecord, SourceSliceRow
from .checkpoints import CHECKPOINT_INTERVAL, write_checkpoint


class UnmappedValueError(Exception):
    def __init__(self, lookup_name: str, source_value: str) -> None:
        self.lookup_name = lookup_name
        self.source_value = source_value
        super().__init__(f"unmapped value '{source_value}' for lookup '{lookup_name}'")


def apply_bindings(
    headers: list[str],
    row_csv: str,
    bindings: list[dict[str, Any]],
) -> dict[str, str]:
    reader = csv.reader(io.StringIO(row_csv))
    values = next(reader)
    source_map = dict(zip(headers, values))
    return {
        b["destination_field"]: source_map.get(b["source_field"], "")
        for b in bindings
    }


def apply_lookup_translation(
    row: dict[str, str],
    lookup_fields: list[tuple[str, str]],
    lookups: dict[str, dict[str, str]],
) -> dict[str, str]:
    result = dict(row)
    for dest_field, lookup_name in lookup_fields:
        source_value = result[dest_field]
        value_map = lookups[lookup_name]
        if source_value not in value_map:
            raise UnmappedValueError(lookup_name, source_value)
        result[dest_field] = value_map[source_value]
    return result


def process_rows(
    run: RunRecord,
    rows: list[SourceSliceRow],
    mapping: MappingSnapshot,
    artifact: MappingArtifact,
    db: Session,
) -> None:
    from ..execution.engine import _LookupDeltaNeeded

    bindings: list[dict[str, Any]] = mapping.field_bindings
    headers = [b["source_field"] for b in bindings]
    lookup_fields = [
        (b["destination_field"], b["lookup_name"])
        for b in bindings
        if b.get("lookup_name")
    ]

    lookups: dict[str, dict[str, str]] = {}
    if lookup_fields and run.lookup_snapshot_version:
        for _, lookup_name in lookup_fields:
            if lookup_name not in lookups:
                snap = (
                    db.query(LookupSnapshot)
                    .filter_by(project_id=run.project_id, lookup_name=lookup_name,
                               lookup_snapshot_version=run.lookup_snapshot_version)
                    .first()
                )
                if snap:
                    lookups[lookup_name] = snap.value_map

    mapped: list[dict[str, str]] = list(artifact.mapped_rows or [])

    for row in rows:
        destination_row = apply_bindings(headers, row.row_csv, bindings)
        try:
            destination_row = apply_lookup_translation(destination_row, lookup_fields, lookups)
        except UnmappedValueError as exc:
            write_checkpoint(run, row.row_index, db)
            raise _LookupDeltaNeeded(exc.lookup_name, exc.source_value)

        mapped.append(destination_row)

        if row.row_index > 0 and row.row_index % CHECKPOINT_INTERVAL == 0:
            artifact.mapped_rows = list(mapped)
            write_checkpoint(run, row.row_index, db)
            db.flush()

    artifact.mapped_rows = mapped
    db.flush()
```

```python
from __future__ import annotations

from fastapi import APIRouter, Depends, status
from sqlalchemy.orm import Session

from ..api.deps import get_central_team_user, get_current_user, get_db, AuthApiError
from ..api.schemas import RunCheckpointResponse, RunCreateRequest, RunResponse
from ..db.models import RunCheckpoint, RunRecord, User, new_id
from ..execution.engine import MissingApprovalError, execute_run
from ..management.access import require_project_access
from ..management.platform import record_management_audit

router = APIRouter(prefix="/projects/{project_id}/runs", tags=["runs"])


def _run_response(r: RunRecord) -> RunResponse:
    return RunResponse(
        run_id=r.run_id,
        project_id=r.project_id,
        destination_object_name=r.destination_object_name,
        environment=r.environment,
        status=r.status,
        current_stage=r.current_stage,
        source_slice_version=r.source_slice_version,
        mapping_snapshot_version=r.mapping_snapshot_version,
        lookup_snapshot_version=r.lookup_snapshot_version,
        code_generation_input_snapshot_version=r.code_generation_input_snapshot_version,
        codegen_artifact_id=r.codegen_artifact_id,
        knowledge_freeze_version=r.knowledge_freeze_version,
        pause_metadata=r.pause_metadata,
        completion_metadata=r.completion_metadata,
        created_at=r.created_at,
        updated_at=r.updated_at,
    )


def _get_run(db: Session, project_id: str, run_id: str) -> RunRecord:
    run = db.get(RunRecord, run_id)
    if run is None or run.project_id != project_id:
        raise AuthApiError("run_not_found", "Run not found.", 404)
    return run


@router.post("", response_model=RunResponse, status_code=status.HTTP_201_CREATED)
def create_run(
    project_id: str,
    body: RunCreateRequest,
    actor: User = Depends(get_central_team_user),
    db: Session = Depends(get_db),
) -> RunResponse:
    run = RunRecord(
        run_id=new_id(),
        project_id=project_id,
        destination_object_name=body.destination_object_name,
        source_definition_reference=body.source_definition_id,
        environment=body.environment,
        status="queued",
    )
    db.add(run)
    record_management_audit(db, project_id=project_id, actor_user_id=actor.user_id,
                            event_type="run.created",
                            payload={"destination_object_name": body.destination_object_name})
    db.commit()
    db.refresh(run)
    return _run_response(run)


@router.get("", response_model=list[RunResponse])
def list_runs(
    project_id: str,
    actor: User = Depends(get_current_user),
    db: Session = Depends(get_db),
) -> list[RunResponse]:
    require_project_access(db, user=actor, project_id=project_id)
    runs = db.query(RunRecord).filter_by(project_id=project_id).all()
    return [_run_response(r) for r in runs]


@router.get("/{run_id}", response_model=RunResponse)
def get_run(
    project_id: str,
    run_id: str,
    actor: User = Depends(get_current_user),
    db: Session = Depends(get_db),
) -> RunResponse:
    require_project_access(db, user=actor, project_id=project_id)
    return _run_response(_get_run(db, project_id, run_id))


@router.post("/{run_id}/launch", response_model=RunResponse)
def launch_run(
    project_id: str,
    run_id: str,
    actor: User = Depends(get_central_team_user),
    db: Session = Depends(get_db),
) -> RunResponse:
    run = _get_run(db, project_id, run_id)
    if run.status != "queued":
        raise AuthApiError("run_not_launchable",
                           f"Run is in '{run.status}' status; only queued runs can be launched.", 409)
    try:
        execute_run(run.run_id, db)
    except MissingApprovalError as exc:
        raise AuthApiError("missing_approval", str(exc), 409)
    db.refresh(run)
    return _run_response(run)


@router.post("/{run_id}/pause", response_model=RunResponse)
def pause_run(
    project_id: str,
    run_id: str,
    actor: User = Depends(get_central_team_user),
    db: Session = Depends(get_db),
) -> RunResponse:
    run = _get_run(db, project_id, run_id)
    if run.status != "running":
        raise AuthApiError("run_not_pausable",
                           f"Run is in '{run.status}' status; only running runs can be paused.", 409)
    run.status = "paused"
    run.pause_metadata = {"reason": "manual"}
    db.commit()
    db.refresh(run)
    return _run_response(run)


@router.post("/{run_id}/resume", response_model=RunResponse)
def resume_run(
    project_id: str,
    run_id: str,
    actor: User = Depends(get_central_team_user),
    db: Session = Depends(get_db),
) -> RunResponse:
    run = _get_run(db, project_id, run_id)
    if run.status not in ("paused", "awaiting_approval"):
        raise AuthApiError("run_not_resumable",
                           f"Run is in '{run.status}' status; cannot resume.", 409)
    run.status = "queued"
    run.pause_metadata = None
    run.resume_metadata = {"resumed_by": actor.user_id}
    db.commit()
    try:
        execute_run(run.run_id, db)
    except MissingApprovalError as exc:
        raise AuthApiError("missing_approval", str(exc), 409)
    db.refresh(run)
    return _run_response(run)


@router.get("/{run_id}/checkpoints", response_model=list[RunCheckpointResponse])
def list_checkpoints(
    project_id: str,
    run_id: str,
    actor: User = Depends(get_current_user),
    db: Session = Depends(get_db),
) -> list[RunCheckpointResponse]:
    require_project_access(db, user=actor, project_id=project_id)
    _get_run(db, project_id, run_id)
    cps = (db.query(RunCheckpoint).filter_by(run_id=run_id)
           .order_by(RunCheckpoint.created_at).all())
    return [
        RunCheckpointResponse(
            run_checkpoint_id=c.run_checkpoint_id,
            run_id=c.run_id,
            current_stage=c.current_stage,
            current_object=c.current_object,
            approved_snapshots=c.approved_snapshots,
            last_completed_checkpoint_boundary=c.last_completed_checkpoint_boundary,
            pause_reason=c.pause_reason,
            created_at=c.created_at,
        )
        for c in cps
    ]
```
