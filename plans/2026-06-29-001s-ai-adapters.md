# AI Adapter Layer — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add the `migrations_engine.ai` package with a YAML-backed config loader, `AIAdapter` protocol, `AnthropicAdapter` and `OpenAIAdapter` implementations, and a `get_adapter(task)` factory. All subsequent AI-calling stages import from here.

**Architecture:** Model slots and provider key references live in `engine/config/engine.yaml` with `${VAR}` env substitution (mirroring the migrations project). The factory resolves the model ID for the requested task slot, infers the provider from the model ID prefix, and returns a configured adapter. Provider API keys are read lazily from the environment at call time.

**Tech Stack:** Python ≥ 3.11, `anthropic` SDK, `openai` SDK, `pyyaml`, Pydantic v2, pytest (no live API calls in tests).

## Global Constraints

- `from __future__ import annotations` at top of every file
- No model slots or provider keys added to the Pydantic `Settings` class — YAML only
- Tests use fixture YAML files and mock both SDKs — never call real APIs
- Config path resolution: `CONFIG_PATH` env var → `engine/config/engine.yaml` → `config/engine.yaml` → `/app/config/engine.yaml`

```python
from __future__ import annotations

import os
import re
from dataclasses import dataclass
from functools import lru_cache
from pathlib import Path
from typing import cast

import yaml

_ENV_PATTERN = re.compile(r"\$\{([A-Z0-9_]+)\}")


class ConfigurationError(Exception):
    """Raised when config is missing, invalid, or references an unset env var."""


@dataclass(frozen=True)
class PlatformModelConfig:
    planning: str
    review: str
    implementation: str


@dataclass(frozen=True)
class MigrationModelConfig:
    pii_review: str
    field_mapping: str
    script_generation: str
    script_correction: str


@dataclass(frozen=True)
class ProviderConfig:
    anthropic_api_key_env: str
    openai_api_key_env: str


@dataclass(frozen=True)
class AIConfig:
    models: PlatformModelConfig
    migration_models: MigrationModelConfig
    providers: ProviderConfig


@lru_cache(maxsize=1)
def get_ai_config(config_path: Path | str | None = None) -> AIConfig:
    path = _resolve_config_path(config_path)
    if not path.exists():
        raise ConfigurationError(f"config file not found: {path}")
    try:
        raw = yaml.safe_load(path.read_text(encoding="utf-8"))
    except yaml.YAMLError as exc:
        raise ConfigurationError(f"invalid YAML in {path}: {exc}") from exc
    if not isinstance(raw, dict):
        raise ConfigurationError(f"expected mapping at top level in {path}")
    substituted = _substitute_env_values(cast(dict, raw))
    return _parse_config(substituted)


def _resolve_config_path(config_path: Path | str | None) -> Path:
    if config_path is not None:
        return Path(config_path)
    env_path = os.environ.get("CONFIG_PATH")
    if env_path:
        return Path(env_path)
    candidates = (
        Path.cwd() / "engine" / "config" / "engine.yaml",
        Path.cwd() / "config" / "engine.yaml",
        Path("/app/config/engine.yaml"),
    )
    for candidate in candidates:
        if candidate.exists():
            return candidate
    return candidates[0]


def _substitute_env_values(value: object) -> object:
    if isinstance(value, str):
        return _ENV_PATTERN.sub(_replace_placeholder, value)
    if isinstance(value, dict):
        return {k: _substitute_env_values(v) for k, v in value.items()}
    if isinstance(value, list):
        return [_substitute_env_values(item) for item in value]
    return value


def _replace_placeholder(match: re.Match[str]) -> str:
    var = match.group(1)
    if var not in os.environ:
        raise ConfigurationError(f"environment variable not set: {var}")
    return os.environ[var]


def _require_mapping(value: object, section: str) -> dict:
    if not isinstance(value, dict):
        raise ConfigurationError(f"expected mapping for section: {section}")
    return cast(dict, value)


def _require_str(mapping: dict, key: str, section: str) -> str:
    value = mapping.get(key)
    if not isinstance(value, str) or not value.strip():
        raise ConfigurationError(f"missing or empty {section}.{key}")
    return value.strip()


def _parse_config(raw: dict) -> AIConfig:
    models_raw = _require_mapping(raw.get("models"), "models")
    migration_raw = _require_mapping(raw.get("migration"), "migration")
    migration_models_raw = _require_mapping(migration_raw.get("models"), "migration.models")
    providers_raw = _require_mapping(raw.get("providers"), "providers")

    return AIConfig(
        models=PlatformModelConfig(
            planning=_require_str(models_raw, "planning", "models"),
            review=_require_str(models_raw, "review", "models"),
            implementation=_require_str(models_raw, "implementation", "models"),
        ),
        migration_models=MigrationModelConfig(
            pii_review=_require_str(migration_models_raw, "pii_review", "migration.models"),
            field_mapping=_require_str(migration_models_raw, "field_mapping", "migration.models"),
            script_generation=_require_str(migration_models_raw, "script_generation", "migration.models"),
            script_correction=_require_str(migration_models_raw, "script_correction", "migration.models"),
        ),
        providers=ProviderConfig(
            anthropic_api_key_env=_require_str(providers_raw, "anthropic_api_key_env", "providers"),
            openai_api_key_env=_require_str(providers_raw, "openai_api_key_env", "providers"),
        ),
    )
```

```python
from __future__ import annotations

import json
from typing import TypeVar

import anthropic

from pydantic import BaseModel

from .adapter import AICallError

T = TypeVar("T", bound=BaseModel)


class AnthropicAdapter:
    def __init__(self, api_key: str, model: str) -> None:
        self._client = anthropic.Anthropic(api_key=api_key)
        self._model = model

    @property
    def model_id(self) -> str:
        return self._model

    def call(self, system: str, user: str, response_model: type[T]) -> T:
        schema_note = (
            f"\n\nRespond with a JSON object matching this schema:\n"
            f"{json.dumps(response_model.model_json_schema(), indent=2)}\n"
            f"Output only the JSON object, no explanation."
        )
        try:
            response = self._client.messages.create(
                model=self._model,
                max_tokens=4096,
                system=system + schema_note,
                messages=[{"role": "user", "content": user}],
            )
        except anthropic.APIError as exc:
            raise AICallError(str(exc)) from exc
        return response_model.model_validate_json(response.content[0].text)
```

```python
from __future__ import annotations

import os
from functools import lru_cache
from pathlib import Path

from .adapter import AIAdapter, ConfigurationError
from .anthropic_adapter import AnthropicAdapter
from .config import AIConfig, get_ai_config
from .openai_adapter import OpenAIAdapter

_SLOT_MAP = {
    "planning":          lambda c: c.models.planning,
    "review":            lambda c: c.models.review,
    "implementation":    lambda c: c.models.implementation,
    "pii_review":        lambda c: c.migration_models.pii_review,
    "field_mapping":     lambda c: c.migration_models.field_mapping,
    "script_generation": lambda c: c.migration_models.script_generation,
    "script_correction": lambda c: c.migration_models.script_correction,
}


@lru_cache(maxsize=32)
def get_adapter(config_path: Path | str | None = None, task: str = "planning") -> AIAdapter:
    """Return a configured AIAdapter for the given task slot.

    Provider is inferred from the model ID:
      "claude-*" or "anthropic/*"  → AnthropicAdapter
      "gpt-*" or "o1-*"           → OpenAIAdapter
    """
    config = get_ai_config(config_path)
    slot_fn = _SLOT_MAP.get(task)
    if slot_fn is None:
        raise ConfigurationError(
            f"unknown task slot: {task!r}. "
            f"Valid slots: {', '.join(_SLOT_MAP)}"
        )
    model_id = slot_fn(config)
    return _build_adapter(config, model_id)


def _build_adapter(config: AIConfig, model_id: str) -> AIAdapter:
    if model_id.startswith("claude-") or model_id.startswith("anthropic/"):
        key_env = config.providers.anthropic_api_key_env
        key = os.environ.get(key_env)
        if not key:
            raise ConfigurationError(
                f"{key_env} is not set. "
                "Set the environment variable before calling AI-dependent stages."
            )
        return AnthropicAdapter(api_key=key, model=model_id)

    if model_id.startswith("gpt-") or model_id.startswith("o1-"):
        key_env = config.providers.openai_api_key_env
        key = os.environ.get(key_env)
        if not key:
            raise ConfigurationError(
                f"{key_env} is not set. "
                "Set the environment variable before calling AI-dependent stages."
            )
        return OpenAIAdapter(api_key=key, model=model_id)

    raise ConfigurationError(
        f"cannot infer provider from model ID: {model_id!r}. "
        "Expected 'claude-*', 'anthropic/*', 'gpt-*', or 'o1-*'."
    )
```
