## Gap 3 — globals.css Tailwind v4 utility registration

### Problem

Tailwind v4 generates utility classes from `@theme` entries. `:root` CSS variables alone
do not generate `bg-*`, `text-*`, or `border-*` utilities. The existing hand-written classes
(`.text-primary`, `.bg-surface`, etc.) only cover a subset. Missing utilities whose opacity
modifier form (`bg-error/10`, `border-error/30`) is used in JSX:

| Class used in JSX | Currently generated? |
|---|---|
| `bg-primary` | ✗ (only `.bg-primary` hand-written — but NOT generated by Tailwind, so JIT purging may drop it) |
| `bg-error`, `bg-error/10` | ✗ |
| `text-error`, `border-error`, `border-error/30` | ✗ |
| `text-warning` | ✗ |
| `bg-primary-container` | ✗ |
| `text-on-primary-container` | ✗ |
| `text-neutral` | ✗ |
| `bg-surface-dim` | ✗ |

### Fix

Add `@theme` block immediately after `@import "tailwindcss"`. Tailwind v4 reads `@theme`
and auto-generates all `bg-*`, `text-*`, `border-*`, and modifier variants.

```css
@import "tailwindcss";

@theme {
  --color-primary: #4f46e5;
  --color-on-primary: #ffffff;
  --color-primary-container: #eef2ff;
  --color-on-primary-container: #3730a3;
  --color-surface: #f8f9fa;
  --color-surface-container: #ffffff;
  --color-surface-dim: #d9dadb;
  --color-error: #ef4444;
  --color-warning: #f59e0b;
  --color-success: #10b981;
  --color-neutral: #6b7280;
  --color-outline-variant: #f3f4f5;
  --color-info: #3b82f6;
}

:root {
  /* ... existing :root block unchanged ... */
}
```

Keep all existing hand-written classes. They take precedence and are harmless duplicates.

```python
def list_projects(
    db: Session,
    *,
    actor: User,
    include_archived: bool = False,
) -> list[ProjectResponse]:
    stmt = select(ProjectRegistry, ProjectDefinition).join(
        ProjectDefinition,
        ProjectRegistry.definition_id == ProjectDefinition.definition_id,
    )
    if not include_archived:
        stmt = stmt.where(ProjectRegistry.archived_at.is_(None))
    stmt = stmt.where(ProjectRegistry.soft_deleted_at.is_(None))

    if actor.role == PROJECT_STAKEHOLDER_ROLE:
        stmt = stmt.join(
            ProjectMembership,
            (ProjectMembership.project_id == ProjectRegistry.project_id)
            & (ProjectMembership.user_id == actor.user_id),
        )

    stmt = stmt.order_by(ProjectRegistry.name)
    rows = db.execute(stmt).all()

    project_ids = [registry.project_id for registry, _ in rows]
    run_summary_by_project = _load_latest_run_summaries(db, project_ids)

    return [
        _project_response(registry, definition, run_summary_by_project.get(registry.project_id))
        for registry, definition in rows
    ]


def _load_latest_run_summaries(
    db: Session,
    project_ids: list[str],
) -> dict[str, LatestRunSummary]:
    if not project_ids:
        return {}

    latest_sub = (
        select(
            RunRecord.project_id,
            func.max(RunRecord.created_at).label("max_created_at"),
        )
        .where(RunRecord.project_id.in_(project_ids))
        .group_by(RunRecord.project_id)
        .subquery()
    )

    stmt = (
        select(RunRecord, SourceDefinition.source_type)
        .join(
            latest_sub,
            and_(
                RunRecord.project_id == latest_sub.c.project_id,
                RunRecord.created_at == latest_sub.c.max_created_at,
            ),
        )
        .outerjoin(
            SourceDefinition,
            RunRecord.source_definition_reference == SourceDefinition.source_definition_id,
        )
    )

    result: dict[str, LatestRunSummary] = {}
    for run, source_type in db.execute(stmt).all():
        result[run.project_id] = LatestRunSummary(
            current_stage=run.current_stage,
            run_status=run.status,
            source_type=source_type,
            stage_entered_at=run.updated_at,
        )
    return result
```

### Component — `web/components/projects/StageTimeline.tsx`

Grid layout matching mockmigration: `grid-cols-2 md:grid-cols-4 lg:grid-cols-9`. Current
stage gets `ring-4 ring-primary/10 animate-pulse`. Past stages use solid emerald fill with
step number (not a checkmark). Current stage shows "N days in stage" subtitle in amber.

```tsx
"use client";

export const STAGE_SEQUENCE = [
  { key: "source_intake",   label: "Intake" },
  { key: "analysis",        label: "Analysis" },
  { key: "mapping",         label: "Mapping" },
  { key: "gate_1_pending",  label: "Gate 1" },
  { key: "gate_2_pending",  label: "Gate 2" },
  { key: "gate_2_approved", label: "Gate 2 ✓" },
  { key: "codegen",         label: "Codegen" },
  { key: "reconciliation",  label: "Reconcile" },
  { key: "complete",        label: "Complete" },
] as const;

export interface StageTimelineProps {
  currentStage: string | null;
  stageEnteredAt: string | null;
}

export function StageTimeline({ currentStage, stageEnteredAt }: StageTimelineProps) {
  const currentIndex = currentStage
    ? STAGE_SEQUENCE.findIndex((s) => s.key === currentStage)
    : -1;

  const daysInStage = stageEnteredAt
    ? Math.floor((Date.now() - new Date(stageEnteredAt).getTime()) / (1000 * 60 * 60 * 24))
    : null;

  return (
    <div className="rounded-2xl border border-outline-variant bg-surface-container p-6 shadow-sm">
      <h3 className="text-xs font-bold text-neutral uppercase tracking-widest font-mono mb-4">
        Governed Lifecycle Stepper
      </h3>
      <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-9 gap-4">
        {STAGE_SEQUENCE.map((stage, index) => {
          const isPast = currentIndex > index;
          const isCurrent = currentIndex === index;

          return (
            <div key={stage.key} className="flex flex-col items-center text-center space-y-2">
              <div
                className={`w-7 h-7 rounded-full flex items-center justify-center font-mono text-xs font-bold border-2 transition ${
                  isCurrent
                    ? "bg-primary text-white border-primary ring-4 ring-primary/10 animate-pulse"
                    : isPast
                      ? "bg-emerald-500 text-slate-950 border-emerald-500"
                      : "bg-surface text-slate-400 border-outline-variant"
                }`}
              >
                {index + 1}
              </div>
              <div className="space-y-0.5">
                <p
                  className={`text-[10px] font-bold truncate max-w-[80px] uppercase font-mono ${
                    isCurrent ? "text-primary" : isPast ? "text-emerald-600" : "text-slate-400"
                  }`}
                >
                  {stage.label}
                </p>
                {isCurrent && daysInStage !== null && (
                  <p className="text-[9px] text-amber-500 font-semibold font-mono">
                    {daysInStage}d in stage
                  </p>
                )}
              </div>
            </div>
          );
        })}
      </div>
      {!currentStage && (
        <p className="mt-4 text-[10px] font-mono text-neutral">No run started yet.</p>
      )}
    </div>
  );
}
```

## Pitfalls

- `LatestRunSummary | None = None` on `ProjectResponse` — Pydantic v2 requires the `= None`
  default to avoid breaking existing callers that don't pass it (e.g., `get_project`).
- The `_load_latest_run_summaries` subquery uses `func.max(RunRecord.created_at)` as a
  tiebreaker. If two runs share the exact same `created_at`, only one is returned — this
  edge case is acceptable since it's a dashboard summary field.
- `SourceDefinition.source_type` is joined via `outerjoin` — if the run's
  `source_definition_reference` is null, `source_type` will be `None` in the result.
  Handle with `source_type` nullable in `LatestRunSummary`.
- Days in Stage calculation: uses `Date.now()` — this means tests that render the days column
  will get a live number. Avoid asserting the exact days count in tests; just assert the
  column header is present.
- `STAGE_SEQUENCE` uses `"gate_2_approved"` as a distinct step from `"gate_2_pending"`.
  If the engine later uses different string values, update only the `key` field in
  `STAGE_SEQUENCE` — the UI labels are independent.
- Do NOT remove Goal, Target DB, or Environments columns without confirming they're truly
  absent from ui.md. Re-read ui.md lines 68–77 before finalizing. (They are not listed there,
  so removing them is correct — but double-check.)
