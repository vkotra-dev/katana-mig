# Runs UI — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Runs list page (`/runs`), run detail page (`/runs/[id]`) with five tabs and polling, and a launch-run dialog — following the stitch mockups at `docs/design/stitch/03-launch-run.md`, `04-run-detail.md`, and `05-runs-list.md` exactly.

**Architecture:** Next.js 15 App Router client components. `runs-api.ts` wraps all fetch calls. Run detail polls every 10 seconds while status is `running`; stops at terminal status. Role-gating (Resume button, Launch button) via `getUiSession()`.

**Depends on:** 001t (runs backend — API endpoints must exist), 001p (project CRUD UI — Topbar/session pattern to follow).

**Tech Stack:** Next.js 15, React 19, Tailwind CSS v4, Vitest 3, @testing-library/react 16.

## Global Constraints

- No invented vocabulary — use exact terms from `docs/domain/runs.md` and stitch mockups
- Monospace IDs: `font-mono` class (JetBrains Mono is already in the design system)
- Every ID column has a copy-to-clipboard icon
- Status chip colors: gray=queued, blue=running, amber=paused/awaiting_approval, green=completed, red=failed
- `getUiSession()` for role; `central_team` only for Launch/Resume/Pause actions
- Auto-poll interval: 10 000 ms; stops when status is `completed` | `failed`

```ts
const BASE = process.env.NEXT_PUBLIC_API_URL ?? "http://127.0.0.1:8000";

export type RunStatus =
  | "queued" | "running" | "paused" | "completed" | "failed" | "awaiting_approval";

export interface RunRecord {
  run_id: string;
  project_id: string;
  destination_object_name: string;
  environment: string | null;
  status: RunStatus;
  current_stage: string | null;
  source_slice_version: string | null;
  mapping_snapshot_version: string | null;
  lookup_snapshot_version: string | null;
  code_generation_input_snapshot_version: string | null;
  codegen_artifact_id: string | null;
  knowledge_freeze_version: string | null;
  pause_metadata: Record<string, unknown> | null;
  completion_metadata: Record<string, unknown> | null;
  created_at: string;
  updated_at: string;
}

export interface RunCheckpoint {
  run_checkpoint_id: string;
  run_id: string;
  current_stage: string | null;
  current_object: string | null;
  approved_snapshots: Record<string, string> | null;
  last_completed_checkpoint_boundary: string | null;
  pause_reason: string | null;
  created_at: string;
}

export interface RunCreateInput {
  destination_object_name: string;
  source_definition_id: string;
  environment?: string;
}

function authHeaders(token: string): Record<string, string> {
  return { Authorization: `Bearer ${token}` };
}

async function handle<T>(res: Response): Promise<T> {
  if (res.ok) return res.json() as Promise<T>;
  let code = "api_error";
  try { const b = await res.json(); code = b?.error?.code ?? code; } catch {}
  throw new Error(code);
}

export const listRuns = (token: string, projectId: string): Promise<RunRecord[]> =>
  fetch(`${BASE}/projects/${projectId}/runs`, { headers: authHeaders(token) })
    .then((r) => handle<RunRecord[]>(r));

export const getRun = (token: string, projectId: string, runId: string): Promise<RunRecord> =>
  fetch(`${BASE}/projects/${projectId}/runs/${runId}`, { headers: authHeaders(token) })
    .then((r) => handle<RunRecord>(r));

export const createRun = (token: string, projectId: string, body: RunCreateInput): Promise<RunRecord> =>
  fetch(`${BASE}/projects/${projectId}/runs`, {
    method: "POST",
    headers: { ...authHeaders(token), "Content-Type": "application/json" },
    body: JSON.stringify(body),
  }).then((r) => handle<RunRecord>(r));

export const launchRun = (token: string, projectId: string, runId: string): Promise<RunRecord> =>
  fetch(`${BASE}/projects/${projectId}/runs/${runId}/launch`, {
    method: "POST", headers: authHeaders(token),
  }).then((r) => handle<RunRecord>(r));

export const pauseRun = (token: string, projectId: string, runId: string): Promise<RunRecord> =>
  fetch(`${BASE}/projects/${projectId}/runs/${runId}/pause`, {
    method: "POST", headers: authHeaders(token),
  }).then((r) => handle<RunRecord>(r));

export const resumeRun = (token: string, projectId: string, runId: string): Promise<RunRecord> =>
  fetch(`${BASE}/projects/${projectId}/runs/${runId}/resume`, {
    method: "POST", headers: authHeaders(token),
  }).then((r) => handle<RunRecord>(r));

export const listCheckpoints = (token: string, projectId: string, runId: string): Promise<RunCheckpoint[]> =>
  fetch(`${BASE}/projects/${projectId}/runs/${runId}/checkpoints`, { headers: authHeaders(token) })
    .then((r) => handle<RunCheckpoint[]>(r));
```

```tsx
"use client";

import { useState } from "react";
import { createRun, launchRun, type RunRecord } from "../../lib/runs-api";

export interface LaunchRunDialogProps {
  open?: boolean;
  token: string;
  onLaunched: (run: RunRecord) => void;
  onClose: () => void;
}

export function LaunchRunDialog({ open, token, onLaunched, onClose }: LaunchRunDialogProps) {
  const [objectName, setObjectName] = useState("");
  const [sourceId, setSourceId] = useState("");
  const [environment, setEnvironment] = useState("");
  const [submitting, setSubmitting] = useState(false);
  const [error, setError] = useState<string | null>(null);

  if (!open) return null;

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setSubmitting(true);
    setError(null);
    try {
      const run = await createRun(token, "", {
        destination_object_name: objectName,
        source_definition_id: sourceId,
        environment: environment || undefined,
      });
      await launchRun(token, run.project_id, run.run_id);
      onLaunched(run);
    } catch (err) {
      setError(err instanceof Error ? err.message : "Unexpected error");
    } finally {
      setSubmitting(false);
    }
  }

  return (
    <div role="dialog" aria-modal="true" aria-label="Launch Run"
         className="fixed inset-0 z-50 flex items-center justify-center bg-black/30">
      <div className="w-full max-w-md rounded-xl border border-outline-variant bg-surface-container p-6 shadow-lg">
        <h2 className="mb-4 text-title-md font-semibold">Launch Run</h2>
```

```tsx
"use client";

import { use, useEffect, useState, useRef } from "react";
import { Topbar } from "../../../components/Topbar";
import { RunStatusChip } from "../../../components/runs/RunStatusChip";
import { getRun, resumeRun, listCheckpoints, type RunRecord, type RunCheckpoint } from "../../../lib/runs-api";
import { getUiSession } from "../../../lib/session";
import type { SessionRole } from "../../../lib/session";

type Tab = "overview" | "snapshots" | "checkpoints" | "timeline" | "reconciliation";
const TERMINAL = new Set(["completed", "failed"]);
const POLL_MS = 10_000;

function VersionPill({ label, value }: { label: string; value: string | null }) {
  if (!value) return <span className="text-xs text-neutral">—</span>;
  return (
    <span className="inline-flex items-center gap-1 rounded border border-outline-variant bg-surface-container px-2 py-0.5 font-mono text-xs">
      {value}
      <button type="button" onClick={() => navigator.clipboard.writeText(value)}
              className="text-neutral hover:text-primary" aria-label={`Copy ${label}`}>⧉</button>
    </span>
  );
}

export default function RunDetailPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = use(params);
  const session = getUiSession();
  const role: SessionRole = session?.role ?? "read_only_auditor";
  const token = session?.access_token ?? "";

  const [run, setRun] = useState<RunRecord | null>(null);
  const [checkpoints, setCheckpoints] = useState<RunCheckpoint[]>([]);
  const [activeTab, setActiveTab] = useState<Tab>("overview");
  const [loading, setLoading] = useState(true);
  const [lastUpdated, setLastUpdated] = useState<Date | null>(null);
  const [resuming, setResuming] = useState(false);
  const pollRef = useRef<ReturnType<typeof setInterval> | null>(null);

  async function loadRun() {
    const [r, cps] = await Promise.all([
      getRun(token, "projects", id),
      listCheckpoints(token, "projects", id),
    ]);
    setRun(r);
    setCheckpoints(cps);
    setLastUpdated(new Date());
    return r;
  }

  useEffect(() => {
    setLoading(true);
    loadRun()
      .then((r) => {
        if (!TERMINAL.has(r.status)) {
          pollRef.current = setInterval(async () => {
            const updated = await loadRun();
            if (TERMINAL.has(updated.status) && pollRef.current) {
              clearInterval(pollRef.current);
            }
          }, POLL_MS);
        }
      })
      .catch(console.error)
      .finally(() => setLoading(false));
    return () => { if (pollRef.current) clearInterval(pollRef.current); };
  }, [token, id]);

  async function handleResume() {
    if (!run) return;
    setResuming(true);
    try {
      const updated = await resumeRun(token, run.project_id, run.run_id);
      setRun(updated);
    } catch (err) {
      console.error(err);
    } finally {
      setResuming(false);
    }
  }

  const tabs: { id: Tab; label: string }[] = [
    { id: "overview", label: "Overview" },
    { id: "snapshots", label: "Pinned snapshots" },
    { id: "checkpoints", label: "Checkpoints" },
    { id: "timeline", label: "Timeline" },
    { id: "reconciliation", label: "Reconciliation & lineage" },
  ];

  return (
    <main className="flex min-h-screen flex-col bg-surface text-slate-800">
      <Topbar role={role} />
      <section className="mx-auto w-full max-w-[1600px] flex-1 px-6 py-4">
        {loading && <p className="py-8 text-center text-sm text-neutral">Loading…</p>}

        {run && (
          <>
            {/* Header */}
            <div className="mb-4 flex flex-wrap items-start justify-between gap-3">
              <div className="flex flex-col gap-1">
                <div className="flex items-center gap-2">
                  <span className="font-mono text-sm text-neutral">{run.run_id}</span>
                  <button type="button"
                          onClick={() => navigator.clipboard.writeText(run.run_id)}
                          className="text-xs text-neutral hover:text-primary">⧉</button>
                </div>
                <h1 className="text-title-lg font-semibold">{run.destination_object_name}</h1>
                <div className="flex items-center gap-2">
                  <RunStatusChip status={run.status} />
                  {run.environment && (
                    <span className="rounded bg-surface-container px-2 py-0.5 text-xs">
                      {run.environment}
                    </span>
                  )}
                </div>
              </div>
              <div className="flex items-center gap-3">
                <span className="text-xs text-neutral">
                  Last updated: {lastUpdated?.toLocaleTimeString() ?? "—"}
                </span>
                <button type="button" onClick={() => loadRun()} className="btn-ghost text-xs">
                  Refresh
                </button>
                {role === "central_team" && run.status === "paused" && (
                  <button type="button" onClick={handleResume} disabled={resuming}
                          className="btn-primary text-sm disabled:opacity-40">
                    {resuming ? "Resuming…" : "Resume"}
                  </button>
                )}
              </div>
            </div>

            {/* State banners */}
            {(run.status === "paused" || run.status === "awaiting_approval") && run.pause_metadata && (
              <div className="mb-4 rounded border border-amber-300 bg-amber-50 px-4 py-3 text-sm text-amber-800">
                Paused — {String(run.pause_metadata.reason ?? "awaiting approval")}
              </div>
            )}
            {run.status === "failed" && (
              <div className="mb-4 rounded border border-red-300 bg-red-50 px-4 py-3 text-sm text-red-800">
                Run failed
              </div>
            )}

            {/* Tabs */}
            <div className="mb-4 flex gap-6 border-b border-outline-variant">
              {tabs.map((t) => (
                <button key={t.id} type="button" role="tab"
                        aria-label={t.label}
                        onClick={() => setActiveTab(t.id)}
                        className={`pb-2 text-sm font-medium transition ${
                          activeTab === t.id
                            ? "border-b-2 border-primary text-primary"
                            : "text-neutral hover:text-slate-800"
                        }`}>
                  {t.label}
                </button>
              ))}
            </div>

            {/* Overview */}
            {activeTab === "overview" && (
              <dl className="grid grid-cols-2 gap-x-8 gap-y-3 text-sm md:grid-cols-3">
                {[
                  ["Status", <RunStatusChip key="s" status={run.status} />],
                  ["Current stage", run.current_stage ?? "—"],
                  ["Destination object", run.destination_object_name],
                  ["Environment", run.environment ?? "—"],
                  ["Created", new Date(run.created_at).toISOString()],
                  ["Updated", new Date(run.updated_at).toISOString()],
                ].map(([label, value]) => (
                  <div key={String(label)}>
                    <dt className="text-xs font-medium text-neutral">{label}</dt>
                    <dd className="mt-0.5">{value}</dd>
                  </div>
                ))}
              </dl>
            )}

            {/* Pinned snapshots */}
            {activeTab === "snapshots" && (
              <dl className="flex flex-col gap-3 text-sm">
                {[
                  ["Source slice", run.source_slice_version],
                  ["Mapping", run.mapping_snapshot_version],
                  ["Lookup", run.lookup_snapshot_version],
                  ["Codegen artifact", run.codegen_artifact_id],
                  ["Knowledge freeze", run.knowledge_freeze_version],
                ].map(([label, value]) => (
                  <div key={String(label)} className="flex items-center gap-4">
                    <dt className="w-40 text-xs font-medium text-neutral">{label}</dt>
                    <dd><VersionPill label={String(label)} value={value as string | null} /></dd>
                  </div>
                ))}
              </dl>
            )}

            {/* Checkpoints */}
            {activeTab === "checkpoints" && (
              checkpoints.length === 0
                ? <p className="py-6 text-sm text-neutral">No checkpoints yet.</p>
                : (
                  <table className="w-full text-sm">
                    <thead>
                      <tr className="border-b border-outline-variant text-xs font-medium text-neutral">
                        <th className="pb-2 pr-4 text-left">Stage</th>
                        <th className="pb-2 pr-4 text-left">Object</th>
                        <th className="pb-2 pr-4 text-left">Last row</th>
                        <th className="pb-2 pr-4 text-left">Pause reason</th>
                        <th className="pb-2 text-left">Created</th>
                      </tr>
                    </thead>
                    <tbody>
                      {checkpoints.map((cp) => (
                        <tr key={cp.run_checkpoint_id}
                            className="border-b border-outline-variant last:border-0">
                          <td className="py-2 pr-4">{cp.current_stage ?? "—"}</td>
                          <td className="py-2 pr-4">{cp.current_object ?? "—"}</td>
                          <td className="py-2 pr-4 font-mono text-xs">
                            {cp.last_completed_checkpoint_boundary ?? "—"}
                          </td>
                          <td className="py-2 pr-4 text-xs text-neutral">
                            {cp.pause_reason ?? "—"}
                          </td>
                          <td className="py-2 font-mono text-xs text-neutral">
                            {new Date(cp.created_at).toISOString()}
                          </td>
                        </tr>
                      ))}
                    </tbody>
                  </table>
                )
            )}

            {/* Timeline */}
            {activeTab === "timeline" && (
              <p className="py-6 text-sm text-neutral">
                Timeline events will appear here as the run progresses.
              </p>
            )}

            {/* Reconciliation */}
            {activeTab === "reconciliation" && (
              <p className="py-6 text-sm text-neutral">
                Reconciliation view — coming in a future task.
              </p>
            )}
          </>
        )}
      </section>
    </main>
  );
}
```
